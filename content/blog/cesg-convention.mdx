# CESG: Components, Effects, State, Graphs

Keep files organized, predictable, and easy to scan.

## Components
- One responsibility per component
- Extract rendering helpers, prefer pure functions

## Effects
- Isolate side effects at the bottom of the file
- Name effects by intent, not lifecycle

## State
- Model state explicitly; derive what you can
- Prefer reducers for complex transitions

## Graphs
- Data dependencies across components; keep them explicit (props) not implicit (globals)

```tsx
type Status = 'idle' | 'loading' | 'success' | 'error'

export function useFetch<T>(fn: () => Promise<T>) {
  const [status, setStatus] = useState<Status>('idle')
  const [data, setData] = useState<T | null>(null)
  const [error, setError] = useState<Error | null>(null)
  // Effect: fetch on mount
  useEffect(() => {
    let mounted = true
    setStatus('loading')
    fn()
      .then((d) => mounted && (setData(d), setStatus('success')))
      .catch((e) => mounted && (setError(e), setStatus('error')))
    return () => {
      mounted = false
    }
  }, [fn])
  return { status, data, error }
}
```
